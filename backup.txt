# null counts per column 
print("===== NULL COUNT =====")
print(combined.isnull().sum().to_string())
    # to.string removes the dtype at the end of the output

# null counts per column percentage
print('\n===== NULL COUNT % =====')
print((combined.isnull().sum() / len(combined) * 100).round(2).to_string())

# Total duplicate rows
print("\n===== DUPLICATE ROWS =====")

# Visualize null rates as a heatmap

    # all the null values per column as a percentage
null_pct = null_pct[null_pct > 0].sort_values(ascending=False)
    # removes the columns with no null values
